// Importando o módulo 'axios' para fazer requisições HTTP
const axios = require('axios');

    try {

    }
}

// Exemplo de utilização da função

    console.log('Produtos:', products);
});

Código Go:


package main

import (
	"fmt"
	"io/ioutil"
	"net/http"
)

const apiURL = "https://api.stockbland.com"

func getAllProducts() ([]byte, error) {
	response, err := http.Get(apiURL + "/products")
	if err != nil {
		return nil, err
	}
	defer response.Body.Close()

	return ioutil.ReadAll(response.Body)
}

func main() {
	products, err := getAllProducts()
	if err != nil {
		fmt.Println("Erro ao obter produtos:", err)
		return
	}
	fmt.Println("Produtos:", string(products))
}

Código C#:


using System;
using System.Net.Http;
using System.Threading.Tasks;

namespace StockblandExample
{
    class Program
    {
        static async Task Main(string[] args)
        {
            string apiUrl = "https://api.stockbland.com";
            HttpClient client = new HttpClient();

            try
            {
                HttpResponseMessage response = await client.GetAsync($"{apiUrl}/products");
                response.EnsureSuccessStatusCode();
                string responseBody = await response.Content.ReadAsStringAsync();
                Console.WriteLine("Produtos: " + responseBody);
            }
            catch (HttpRequestException e)
            {
                Console.WriteLine($"Erro ao obter produtos: {e.Message}");
            }
        }
    }
}

Código Java:


import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

public class Main {
    public static void main(String[] args) {
        String apiUrl = "https://api.stockbland.com/products";
        try {
            URL url = new URL(apiUrl);
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");
            BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream()));
            String line;
            StringBuffer response = new StringBuffer();
            while ((line = reader.readLine()) != null) {
                response.append(line);
            }
            reader.close();
            System.out.println("Produtos: " + response.toString());
        } catch (IOException e) {
            System.out.println("Erro ao obter produtos: " + e.getMessage());
        }
    }
}
