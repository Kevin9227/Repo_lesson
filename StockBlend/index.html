// Importando o módulo 'axios' para fazer requisições HTTP
const axios = require('axios');

// URL base da API
const API_URL = 'https://api.stockbland.com';

// Função para obter todos os produtos
async function getAllProducts() {
    try {
        const response = await axios.get(`${API_URL}/products`);
        return response.data;
    } catch (error) {
        console.error('Erro ao obter produtos:', error);
    }
}

// Exemplo de utilização da função
getAllProducts().then(products => {
    console.log('Produtos:', products);
});

Código CSS:


package main

import (
	"fmt"
	"io/ioutil"
	"net/http"
)

const apiURL = "https://api.stockbland.com"

func getAllProducts() ([]byte, error) {
	response, err := http.Get(apiURL + "/products")
	if err != nil {
		return nil, err
	}
	defer response.Body.Close()

	return ioutil.ReadAll(response.Body)
}

func main() {
	products, err := getAllProducts()
	if err != nil {
		fmt.Println("Erro ao obter produtos:", err)
		return
	}
	fmt.Println("Produtos:", string(products))
}

Código Go:


import 'dart:convert';
import 'package:http/http.dart' as http;

const apiUrl = 'https://api.stockbland.com';

Future> getAllProducts() async {
  try {
    final response = await http.get(Uri.parse('$apiUrl/products'));
    return json.decode(response.body);
  } catch (e) {
    print('Erro ao obter produtos: $e');
    return [];
  }
}

void main() async {
  final products = await getAllProducts();
  print('Produtos: $products');
}

Código PYTHON:


import requests

# URL base da API
API_URL = 'https://api.stockbland.com'

# Função para obter todos os produtos
def get_all_products():
    try:
        response = requests.get(f'{API_URL}/products')
        return response.json()
    except Exception as e:
        print('Erro ao obter produtos:', e)

# Exemplo de utilização da função
products = get_all_products()
print('Produtos:', products)
